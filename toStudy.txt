WEB SERVER
    # Muestra todos los servicios instalados
    service --status-all
    # Verifica el estado del servicio
    service apache2 status
    # Cambia el estado del servicio
    service apache2 start/restart/stop

    ## Preparando el servicio
    # Instala el packete de apache2
    sudo apt-get install apache2
    # Dehabilitamos el inicio automatico cuando la pc se prende
    sudo update-rc.d apache2 disable
    # Iniciar el servicio apache2
    sudo service apache2 start
    # Configurar firewall para permitir trafico de apache2
    sudo ufw allow apache2

    ## Iniciando la pagina web
    # Ir a la carpera donde deben ir las paginas web
    cd /var/www/html/
    # Crear carpeta para la pagina web
    mkdir paginaweb
    # Entrar a la carpeta de la pagina web
    cd paginaweb/
    # Crear archivo index.html para probar que funciona
    touch index.html
    # Agregar algo de codigo al archivo
    sudo nano index.html
    <h1>HOLA MUNDO</h1>
    # Entrar a la pagina web a travez de un navegador web
    localhost/paginaweb/
    # Esto debería mostrar el mensaje "HOLA MUNDO", por ahora nuestra pagina es local

    ## Crear archivo de configuracion para la pagina web
    # Entrar a la carpeta donde los archivos de configuracion de apache se encuentran
    cd /etc/apache2/sites-available/
    # Duplicar el archivo 000-default.conf con el nombre mipagina.conf

    ## Activar configuracion nueva y desactivar configuracion default
    # Deshabilitar el archivo de configuracion por defecto
    sudo a2dissite 000-default.conf
    # Habilitar el archivo de configuracion que hemos creado para nuestra pagina
    sudo a2ensite mipagina.conf
    # Reiniciar el servicio apache para que los cambios surjan efecto
    sudo service apache2 restart

    ## Modificar archivo de DNS
    # Entrar a la carpeta del archivo hosts
    cd /etc/
    # Modificar archivo hosts y agrega tu ip seguido del ServerName que le pusiste a cada VirtualHost
    sudo nano hosts

    ## Preparando el servicio
    # Instalar
    sudo apt-get install mysql-server
    # Configurar firewall para permitir trafico de mysql
    sudo ufw allow mysql

    ## Configurar mysql para permitir acceso remoto
    # Entrar a la carpeta de configuracion
    cd /etc/mysql
    # Revisar el archivo my.cnf y buscar en las carpetas incluidas el archivo mysqld.cnf
    # Abrimos el archivo mysql.conf y buscamos la linea que tenga bind-address
    # Cambiamos la ip de bind-address y ponemos la nuestra, ahora el servidor mysql esta "amarrado" a nuestra ip

    ## Configurar a base de datos
    # Entrar como root a la base de datos
    sudo mysql -u root
    # Crear un nuevo usuario que sera remoto con acceso a mysql
    CREATE USER 'nombre'@'dominio' IDENTIFIED BY 'contraseña';
    # Darle privilegios de administrador
    GRANT ALL PRIVILEGES ON databasename.tablename TO 'usuario'@'dominio';
    # Darle privilegios especificos
    GRANT SELECT ON databasename TO 'usuario'@'dominio';
    # Extra: en lugar de poner el nombre de la base de datos, podemos poner * que significa que tiene acceso a todos
    # Ejm: GRANT ALL PRIVILEGES ON *.* TO 'usuario'@'dominio';

    # Entrar a la base de datos
    mysql -u usuario -p -h 192.168.43.xxx
    # -u: usuario; -p: indica que el usuario se logeara con contraseña; -h: indica la ip del servidor

    # Usar una base de datos
    USE databasename;
    # Mostrar informacion de una tabla
    SELECT * FROM nombretabla;
    # Modificar un valor, si if_campo es igual a if_valor, entonces campo=valor
    UPDATE tabla SET campo=valor WHERE if_campo=if_valor
